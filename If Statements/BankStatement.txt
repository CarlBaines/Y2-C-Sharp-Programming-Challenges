        private void txtBankBalance_TextChanged(object sender, EventArgs e)
        {
            //Variables

            double bankBalance;

            double withdrawAmount;

            double newBalance;

           

            //Input bank balance

            bankBalance = double.Parse(txtBankBalance.Text);

            // Check if something has actually been entered.

            if (txtWithdrawAmount.Text.Length == 0) { return; }

            //Input amount you would like to withdraw

            withdrawAmount = double.Parse(txtWithdrawAmount.Text);

            //Calculate new balance and whether the withdraw is allowed or not.

            if (withdrawAmount <= bankBalance)
            {
                newBalance = bankBalance - withdrawAmount;
                txtNewBalance.Text = newBalance.ToString();
                txtMessage.Text = "N/A";
            }
            else
            {
                txtNewBalance.Text = "N/A";
                txtMessage.Text = "Withdrawal is not allowed - you will be overdrawn";
            }
    
            
            
        }
